-- 家庭帳本表
CREATE TABLE families (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  invite_code TEXT UNIQUE NOT NULL,
  invite_expires_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 家庭成員表
CREATE TABLE family_members (
  id BIGSERIAL PRIMARY KEY,
  family_id UUID NOT NULL REFERENCES families(id) ON DELETE CASCADE,
  user_id UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  UNIQUE(family_id, user_id)
);

-- 交易記錄表
CREATE TABLE transactions (
  id BIGSERIAL PRIMARY KEY,
  family_id UUID NOT NULL REFERENCES families(id) ON DELETE CASCADE,
  type TEXT NOT NULL CHECK (type IN ('income','expense')),
  description TEXT NOT NULL,
  amount NUMERIC NOT NULL,
  who TEXT NOT NULL CHECK (who IN ('husband','wife','joint')),
  category TEXT NOT NULL,
  date DATE NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 預算表
CREATE TABLE budgets (
  id BIGSERIAL PRIMARY KEY,
  family_id UUID NOT NULL REFERENCES families(id) ON DELETE CASCADE,
  category TEXT NOT NULL,
  amount NUMERIC NOT NULL,
  UNIQUE(family_id, category)
);

-- 啟用 RLS
ALTER TABLE families ENABLE ROW LEVEL SECURITY;
ALTER TABLE family_members ENABLE ROW LEVEL SECURITY;
ALTER TABLE transactions ENABLE ROW LEVEL SECURITY;
ALTER TABLE budgets ENABLE ROW LEVEL SECURITY;

-- 政策：只能看自己家庭的資料
CREATE POLICY "own_family" ON families FOR ALL USING (
  id IN (SELECT family_id FROM family_members WHERE user_id = auth.uid())
);
CREATE POLICY "own_members" ON family_members FOR ALL USING (user_id = auth.uid());
CREATE POLICY "own_tx" ON transactions FOR ALL USING (
  family_id IN (SELECT family_id FROM family_members WHERE user_id = auth.uid())
) WITH CHECK (
  family_id IN (SELECT family_id FROM family_members WHERE user_id = auth.uid())
);
CREATE POLICY "own_budget" ON budgets FOR ALL USING (
  family_id IN (SELECT family_id FROM family_members WHERE user_id = auth.uid())
) WITH CHECK (
  family_id IN (SELECT family_id FROM family_members WHERE user_id = auth.uid())
);
